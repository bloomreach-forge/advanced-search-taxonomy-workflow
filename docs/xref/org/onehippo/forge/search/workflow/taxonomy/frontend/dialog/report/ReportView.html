38  public class ReportView extends Panel {
39  
40      private static final String REPORT_CONTAINER_ID = "report";
41      private static final AttributeModifier[] NO_ATTRIBUTE_MODIFIERS = new AttributeModifier[0];
42  
43      private TableDefinition<Failure> tableDefinition;
44      private ListDataTable<Failure> dataTable;
45      private ListPagingDefinition pagingDefinition;
46      private final IBrowseService browseService;
47      private final IModel<Report> reportModel;
48  
49      public ReportView(final String id, final IModel<Report> reportModel, final IBrowseService browseService) {
50          super(id, reportModel);
51          this.browseService = browseService;
52          this.reportModel = reportModel;
53          add(new EmptyPanel(REPORT_CONTAINER_ID));
54          setOutputMarkupId(true);
55      }
56  
57      @Override
58      protected void onBeforeRender() {
59          if (reportModel.getObject().hasFailures()) {
60              replaceWithErrorFragment();
61          } else {
62              replaceWithSuccessFragment();
63          }
64          super.onBeforeRender();
65      }
66  
67      private org.apache.wicket.markup.html.basic.Label createLabel(final String resourceKeySuffix) {
68          return new org.apache.wicket.markup.html.basic.Label("message",
69                  new LoadableDetachableModel<String>() {
70                      @Override
71                      protected String load() {
72                          final Report report = reportModel.getObject();
73                          final String resourceKey = report.getAction() + resourceKeySuffix;
74                          return new StringResourceModel(resourceKey, ReportView.this, null, new Object[]{report.getSucceeded()}).getString();
75                      }
76                  });
77      }
78  
79      private void replaceWithSuccessFragment() {
80          replace(new Fragment(REPORT_CONTAINER_ID, "success", this) {
81              {
82                  add(createLabel("-success"));
83              }
84          });
85      }
86  
87      private void replaceWithErrorFragment() {
88          replace(new Fragment(REPORT_CONTAINER_ID, "errors", this) {
89              {
90                  add(createLabel("-error-message"));
91  
92                  final ISortableDataProvider<Failure, String> dataProvider = new SortableDataProvider<Failure>() {
93  
94                      private java.util.List<Failure> failures;
95  
96                      private void load() {
97                          if (failures == null) {
98                              Report report = reportModel.getObject();
99                              failures = report.getFailures();
100                         }
101                     }
102 
103                     @Override
104                     public void detach() {
105                         failures = null;
106                     }
107 
108                     @Override
109                     public Iterator<? extends Failure> iterator(final long first, final long count) {
110                         load();
111                         return failures.subList((int) first, (int) (first + count)).iterator();
112                     }
113 
114                     @Override
115                     public long size() {
116                         load();
117                         return failures.size();
118                     }
119 
120                     @Override
121                     public IModel<Failure> model(Failure failure) {
122                         return new FailureModel(failure);
123                     }
124 
125                 };
126 
127                 List<ListColumn<Failure>> columns = new ArrayList<>();
128 
129                 ListColumn<Failure> column = new ListColumn<>(Model.of(""), "icon");
130                 final DocumentIconAndStateRenderer iconRenderer = new DocumentIconAndStateRenderer();
131                 column.setRenderer(new IListCellRenderer<Failure>() {
132                     @Override
133                     public Component getRenderer(final String id, final IModel<Failure> model) {
134                         return iconRenderer.getRenderer(id, new JcrNodeModel(model.getObject().getNode()));
135                     }
136 
137                     @Override
138                     public IObservable getObservable(final IModel<Failure> model) {
139                         return iconRenderer.getObservable(new JcrNodeModel(model.getObject().getNode()));
140                     }
141                 });
142 
143                 final StateIconAttributeModifier iconAttributeModifier = new StateIconAttributeModifier();
144                 column.setAttributeModifier(new AbstractListAttributeModifier<Failure>() {
145                     @Override
146                     public AttributeModifier[] getCellAttributeModifiers(final IModel<Failure> model) {
147                         final JcrNodeModel nodeModel = new JcrNodeModel(model.getObject().getNode());
148                         return iconAttributeModifier.getCellAttributeModifiers(nodeModel);
149                     }
150                 });
151                 column.setCssClass(DocumentListColumn.ICON.getCssClass());
152 
153                 columns.add(column);
154 
155                 column = new ListColumn<>(new ResourceModel("document-name"), "name");
156                 column.setRenderer(new IListCellRenderer<Failure>() {
157 
158                     private IModel<String> getDisplayName(Failure failure) {
159                         try {
160                             return DocumentUtils.getDocumentNameModel(failure.getNode());
161                         } catch (RepositoryException ignored) {
162                         }
163                         return new NodeNameModel(new JcrNodeModel(failure.getNode()));
164                     }
165 
166                     public Component getRenderer(String id, IModel<Failure> model) {
167                         Failure failure = model.getObject();
168                         return new org.apache.wicket.markup.html.basic.Label(id, getDisplayName(failure));
169                     }
170 
171                     public IObservable getObservable(IModel<Failure> model) {
172                         Failure failure = model.getObject();
173                         IModel<String> displayName = getDisplayName(failure);
174                         if (displayName instanceof IObservable) {
175                             return (IObservable) displayName;
176                         }
177                         return null;
178                     }
179                 });
180                 column.setCssClass(DocumentListColumn.NAME.getCssClass());
181                 columns.add(column);
182 
183                 column = new ListColumn<>(new ResourceModel("reason"), "reason");
184                 column.setRenderer(new IListCellRenderer<Failure>() {
185 
186                     public Component getRenderer(String id, IModel<Failure> model) {
187                         Failure failure = model.getObject();
188                         return new Label(id, new ResourceModel(failure.getReason()));
189                     }
190 
191                     public IObservable getObservable(IModel<Failure> model) {
192                         return null;
193                     }
194                 });
195                 column.setCssClass("doclisting-reason");
196                 columns.add(column);
197 
198                 tableDefinition = new TableDefinition<>(columns, true);
199                 pagingDefinition = new ListPagingDefinition();
200                 pagingDefinition.setPageSize(12);
201 
202                 ListDataTable.TableSelectionListener<Failure> selectionListener = model -> {
203                     final Node node = model.getObject().getNode();
204                     browseService.browse(new JcrNodeModel(node));
205                 };
206                 dataTable = new ListDataTable<Failure>("failed", tableDefinition, dataProvider, selectionListener,
207                         false, pagingDefinition) {
208                     @Override
209                     public boolean isVisible() {
210                         return dataProvider.size() > 0;
211                     }
212                 };
213 
214                 add(dataTable);
215             }
216         });
217     }
